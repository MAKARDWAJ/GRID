# ========== CORE UTILITY FUNCTIONS ==========
def is_numeric(val):
    """Detect numeric values (skip dates, SR.NO, NaN)."""
    if pd.isna(val): return False
    if isinstance(val, (pd.Timestamp, datetime)): return False
    if isinstance(val, str):
        val_str = str(val).strip()
        if '-' in val_str and len(val_str) <= 5: return False
        if val_str.upper() in ['SR.NO', 'SR NO', 'SRNO']: return False
        try: 
            float(val)
            return True
        except: 
            return False
    return isinstance(val, (int, float, np.integer, np.floating))

def safe_int(val):
    """Convert to int safely (return None for invalid)."""
    if isinstance(val, (pd.Timestamp, datetime)): return None
    try: 
        return int(float(val))
    except: 
        return None

def excel_coord(row_idx, col_idx):
    """Convert 0-indexed to Excel A1 notation."""
    col_letter = ''
    col = col_idx
    while col >= 0:
        col_letter = chr(ord('A') + (col % 26)) + col_letter
        col = col // 26 - 1
    return f"{col_letter}{row_idx + 1}"

# ========== DIRECTIONS ==========
def get_directions():
    return {
        "RIGHT": (0, 1), "LEFT": (0, -1),
        "DOWN": (1, 0), "UP": (-1, 0),
        "DOWN_RIGHT": (1, 1), "UP_LEFT": (-1, -1),
        "DOWN_LEFT": (1, -1), "UP_RIGHT": (-1, 1),
    }

def dir_symbol(direction_name):
    symbols = {
        "RIGHT": "→", "LEFT": "←", "DOWN": "↓", "UP": "↑",
        "DOWN_RIGHT": "↘", "UP_LEFT": "↖", "DOWN_LEFT": "↙", "UP_RIGHT": "↗"
    }
    return symbols.get(direction_name, "?")

# ========== PATTERN 1: SINGLE DIGITS ==========
def find_single_patterns(grid, pattern, group_start_col=0):
    rows, cols = grid.shape
    matches = []
    directions = get_directions()
    
    for r in range(rows):
        for c in range(cols):
            for dir_name, (dr, dc) in directions.items():
                seq_digits = []
                pos_seq = []
                
                valid = True
                for k in range(len(pattern)):
                    rr = r + k * dr
                    cc = c + k * dc
                    if 0 <= rr < rows and 0 <= cc < cols:
                        val = grid[rr, cc]
                        if not is_numeric(val):
                            valid = False
                            break
                        digit = safe_int(val) % 10
                        seq_digits.append(digit)
                        pos_seq.append((rr, cc))
                    else:
                        valid = False
                        break
                
                if valid and seq_digits == pattern:
                    cells = [excel_coord(DATA_START_ROW + rp, group_start_col + cp) 
                           for rp, cp in pos_seq]
                    vals = [safe_int(grid[rp, cp]) for rp, cp in pos_seq]
                    matches.append(f"[1-SINGLE] {pattern} {dir_symbol(dir_name)} {vals}: {', '.join(cells)}")
    return matches

def find_vertical_patterns(grid, pattern, group_start_col=0):

    rows, cols = grid.shape

        "DOWN_LEFT": (1, -1)
    }  # Only "forward" directions - no LEFT/UP mirrors
    
    for start_r in range(rows):
        for start_c in range(cols):
            for dir_name, (dr, dc) in primary_directions.items():  # ✅ LIMITED DIRECTIONS
                seq_digits = []
                pos_seq = []
                valid = True
                
                for k in range(len(pattern)):
                    r = start_r + k * dr
                    c = start_c + k * dc
                    
                    if r + 1 >= rows or r < 0 or c < 0 or c >= cols:
                        valid = False
                        break
                    
                    val1 = safe_int(grid[r, c])
                    val2 = safe_int(grid[r + 1, c])
                    
                    if val1 is None or val2 is None:
                        valid = False
                        break
                    
                    total = val1 + val2
                    seq_digits.append(total % 10)
                    pos_seq.append((r, c))
                
                if valid and seq_digits == pattern:

                    sums_display = []
                    for r, c in pos_seq:

                        sums_display.append(str(safe_int(grid[r,c]) + safe_int(grid[r+1,c])))
                    
                    matches.append(f"[2-VERT] {pattern} {dir_symbol(dir_name)} [{','.join(sums_display)}]: {', '.join(cells)}")
    
    return matches

# ========== PATTERN 4: HORIZONTAL 3-DIGIT ==========
def find_horizontal_3digit_sums(grid, pattern, group_start_col=0):

# ========== PATTERN 5: 2-DIGIT BREAKDOWN ==========
def find_2digit_sums(grid, pattern, group_start_col=0):
    rows, cols = grid.shape
    matches = []
    directions = get_directions()
    
    for r in range(rows):
        for c in range(cols):
            for dir_name, (dr, dc) in directions.items():
                seq_digits = []
                pos_seq = []
                
                valid = True
                for k in range(len(pattern)):
                    rr = r + k * dr
                    cc = c + k * dc
                    if 0 <= rr < rows and 0 <= cc < cols:
                        val = safe_int(grid[rr, cc])
                        if val is None or val < 10:
                            valid = False
                            break
                        tens = val // 10
                        units = val % 10
                        total = tens + units
                        digit = total % 10
                        seq_digits.append(digit)
                        pos_seq.append((rr, cc))
                    else:
                        valid = False
                        break
                
                if valid and seq_digits == pattern:
                    cells = [excel_coord(DATA_START_ROW + rp, group_start_col + cp) for rp, cp in pos_seq]
                    breakdown = []
                    for rp, cp in pos_seq:
                        val = safe_int(grid[rp, cp])
                        tens = val // 10
                        units = val % 10
                        total = tens + units
                        breakdown.append(f"{tens}+{units}={total}")
                    matches.append(f"[5-2DIG] {pattern} {dir_symbol(dir_name)} [{','.join(breakdown)}]: {', '.join(cells)}")
    return matches

# ========== DEDUPLICATION FUNCTION ==========
def deduplicate_matches(matches):
    """Remove exact cell set duplicates regardless of pattern/direction"""
    seen_cells = set()
    unique_matches = []
    
    for match in matches:
        # Extract just the cells (ignore pattern/dir)
        cells_part = match.split(': ')[1] if ': ' in match else match
        cells_tuple = tuple(sorted(cells_part.split(', ')))  # Sort for consistency
        
        if cells_tuple not in seen_cells:
            seen_cells.add(cells_tuple)
            unique_matches.append(match)
    
    return unique_matches
